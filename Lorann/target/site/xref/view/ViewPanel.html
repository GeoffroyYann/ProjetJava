32      
33      public void update(IElement[][] tileMap) {
34          this.tileMap = tileMap;
35          this.setSize(this.tileMap[0].length, this.tileMap.length);
36          this.repaint();
37      }
38      
39      public void setSize(int width, int height) {
40          super.setSize((width*32) + this.getInsets().left + this.getInsets().right, (height*32) + this.getInsets().top + this.getInsets().bottom +40);
41          this.viewFrame.setSize(width*32, height*32 + 40);
42      }
43      
44      protected void PaintComponent(final Graphics graphics) {
45          graphics.setColor(Color.black);
46          graphics.fillRect(0, 0, this.getWidth(), this.getHeight());
47          graphics.setColor(Color.yellow);
48          graphics.setFont(new Font(null, Font.BOLD, 20));
49          
50          int scoreIndex = 0;
51          
52          String[][] scores = null;
53          if(this.tileMap != null) {
54              for(int i = 0; i < this.tileMap.length; i++) {
55                  for(int j = 0; i < this.tileMap[0].length; j++) {
56                      BufferedImage image = tileMap[i][j].getImage();
57                      if(image != null)
58                          graphics.drawImage(image, j*32, i*32, null);
59                      else if(tileMap[i][j].getClass().getSimpleName().contains("Title")) {
60                          graphics.drawString("HIGHSCORE", j*32, i*32 + 20);
61                      } else if(tileMap[i][j].getClass().getSimpleName().contains("Score")) {
62                          if(scores == null)
63                              scores = this.viewFrame.getController().getScores();
64                          
65                          if(scoreIndex < scores[0].length) {
66                              graphics.drawString(String.format("%s %s", scores[0][scoreIndex], scores[1][scoreIndex]), j*32 + 5, i*32 + 20);
67                              scoreIndex++;
68                          }
69                      }
70                  }
71              }
72          }
73          graphics.drawString(String.format("SCORE : %d    LEVEL : %d", this.viewFrame.getController().getScore(), this.viewFrame.getController().getLevel()), 10, this.getHeight() - 20);
74      }
