13      public Lorann(Point pos){
14          super("lorann.gif", true, pos);

16      
17      public void move(MobileOrder order, IElement[][] tileMap, IView view){
18          switch (order){
19              case Left:
20                  if(this.direction == MobileOrder.Left && this.pos.y > 0 && tileMap[this.pos.x][this.pos.y - 1].getPermeability()){
21                      this.pos.setLocation(this.pos.getX(), this.pos.getY() - 1);
22                  }
23                  break;
24              case Right:
25                  if(this.direction == MobileOrder.Right && this.pos.y < view.getWidth() / 32 - 1 && tileMap[this.pos.x][this.pos.y + 1].getPermeability()){
26                      this.pos.setLocation(this.pos.getX(), this.pos.getY() + 1);
27                  }
28                  break;
29              case Up:
30                  if(this.direction == MobileOrder.Up && this.pos.x > 0 && tileMap[this.pos.x - 1][this.pos.y].getPermeability()){
31                      this.pos.setLocation(this.pos.getX() - 1, this.pos.getY());
32                  }
33                  break;
34              case Down:
35                  if(this.direction == MobileOrder.Down && this.pos.x < view.getHeight() / 32 - 1 && tileMap[this.pos.x + 1][this.pos.y].getPermeability()){
36                      this.pos.setLocation(this.pos.getX() + 1, this.pos.getY());
37                  }
38                  break;
39          }
40          this.direction = order;
41      }
42  }
